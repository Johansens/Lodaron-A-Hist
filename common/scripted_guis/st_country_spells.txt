scripted_gui = {
	
	st_country_powers = {
	
		context_type = selected_country_context
		
		window_name = "st_subject_powers_container"
		
		visible = {
			ROOT = {
				is_ai = no
				has_country_flag = transfer_enabled_st
			}
		}
		
		effects = {
		
			mark_all_states_click = {
				set_temp_variable = { ROOT.marks_st = this}
				var:marks_st = {
					every_owned_state = {
						limit = {
							NOT = { has_state_flag = marked_for_transfer_st }
						}
						set_state_flag = marked_for_transfer_st
						add_to_variable = { ROOT.marked_states_st = 1 }
						#set_border_war = yes
					}
				}
			}
		
			mark_colonial_states_click  = {
				set_temp_variable = { ROOT.marks_st = this}
				var:marks_st = {
					every_owned_state = {
						limit = {
							NOT = { is_in_home_area = yes }
							NOT = { is_core_of = PREV }
							NOT = { has_state_flag = marked_for_transfer_st }
						}
						set_state_flag = marked_for_transfer_st
						add_to_variable = { ROOT.marked_states_st = 1 }
						#set_border_war = yes
					}
				}
			}
		
			clear_all_markings_click = {
				ROOT = { 
					set_variable = { marked_states_st = 0 }
				}
				every_state = {
					limit = {
						has_state_flag = marked_for_transfer_st
						OR = {
							is_owned_and_controlled_by = ROOT
							any_country = {
								is_ai = yes
								is_subject_of = ROOT
								owns_state = PREV
							}
						}
					}
					clr_state_flag = marked_for_transfer_st
					#set_border_war = no
				}
			}
			
			award_states_click = {
				ROOT = { 
					set_variable = { marked_states_st = 0 }
				}
				set_temp_variable = { ROOT.taker_St = this}
				
				every_state = {
					limit = {
						has_state_flag = marked_for_transfer_st
						OR = {
							is_owned_and_controlled_by = ROOT
							any_country = {
								is_ai = yes
								is_subject_of = ROOT
								owns_state = PREV
							}
						}
					}
					var:taker_St = { transfer_state = PREV }
					clr_state_flag = marked_for_transfer_st
					#set_border_war = no
				}
			}
		}
		
		triggers = {
		
			mark_all_states_click_enabled = {
				any_owned_state = {
					NOT = { has_state_flag = marked_for_transfer_st }
				}
			}
			
			mark_all_states_visible = {
				set_temp_variable = { ROOT.taker_St = this}
				OR = {
					ROOT = {
						tag = var:taker_St
					}
					AND = {
						has_game_rule = {
							rule = allow_subject_transfers
							option = FREE
						}
						any_country = {
							is_ai = yes
							is_subject_of = ROOT
						}
					}
					AND = {
						has_game_rule = {
							rule = allow_subject_transfers
							option = LOW_AUTONOMY_ONLY
						}
						any_country = {
							is_ai = yes
							is_subject_of = ROOT
							compare_autonomy_state < autonomy_colony
						}
					}
				}
			}
		
			mark_colonial_states_click_enabled = {
				any_owned_state = {
					NOT = { is_in_home_area = yes }
					NOT = { has_state_flag = marked_for_transfer_st }
					NOT = { is_core_of = this }
				}
			}
			
			mark_colonial_states_visible = {
				set_temp_variable = { ROOT.taker_St = this}
				OR = {
					ROOT = {
						tag = var:taker_St
					}
					AND = {
						has_game_rule = {
							rule = allow_subject_transfers
							option = FREE
						}
						any_country = {
							is_ai = yes
							is_subject_of = ROOT
						}
					}
					AND = {
						has_game_rule = {
							rule = allow_subject_transfers
							option = LOW_AUTONOMY_ONLY
						}
						any_country = {
							is_ai = yes
							is_subject_of = ROOT
							compare_autonomy_state < autonomy_colony
						}
					}
				}
			}
			
			clear_all_markings_click_enabled = {
				check_variable = { ROOT.marked_states_st > 0 }
			}
		
			award_states_click_enabled = {
				check_variable = { ROOT.marked_states_st > 0 }
				NOT = { 
					ROOT = { 
						all_owned_state = {
							has_state_flag = marked_for_transfer_st
						} 
					}
				}
			}
		
		}
        
	}
}
